14  def reformat_para(para='',left=0,right=72,just=LEFT): 
15      words = para.split() 
16      lines = [] 
17      line  = '' 
18      word = 0 
19      end_words = 0 
20      while not end_words: 
21          if len(words[word]) > right-left: # Handle very long words 
22              line = words[word] 
23              word +=1 
24              if word >= len(words): 
25                  end_words = 1 
26          else:                             # Compose line of words 
27              while len(line)+len(words[word]) <= right-left: 
28                  line += words[word]+' ' 
29                  word += 1 
30                  if word >= len(words): 
31                      end_words = 1 
32                      break 
33          lines.append(line) 
34          line = '' 
35      if just==CENTER: 
36          r, l = right, left 
37          return '\n'.join([' '*left+ln.center(r-l) for ln in lines]) 
38      elif just==RIGHT: 
39          return '\n'.join([line.rjust(right) for line in lines]) 
40      else: # left justify 
41          return '\n'.join([' '*left+line for line in lines])
