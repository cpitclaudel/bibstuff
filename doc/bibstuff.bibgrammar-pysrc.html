30  # EBNF description of a bibtex file 
31  dec = r""" 
32  bibfile              := entry_or_junk* 
33  >entry_or_junk<      := (tb, object) / (tb, junk) 
34  >object<               := entry / macro / preamble / comment_entry 
35  entry                := '@', type , tb,  ( '{' , tb, contents, tb, '}' ) / ( '(' , tb, contents, tb, ')' ) 
36  macro                := '@', type, tb,  ( '{' , tb, macro_contents, tb, '}' ) / ( '(' , tb, macro_contents, tb, ')' ) 
37  preamble             := '@', type, tb,  ( '{' , tb, preamble_contents, tb, '}' ) / ( '(' , tb, preamble_contents, tb, ')' ) 
38  comment_entry        := '@', type, tb, string 
39  >contents<           := key , ',' , fields 
40  >macro_contents<     := fields 
41  >preamble_contents<  := value 
42  type                 := name 
43  key                  := number / name 
44  fields               := tb, field_comma+ , field? 
45  >field_comma<        := field , tb, ',' 
46  field                := tb, name, tb, '=' , tb, value 
47  value                := simple_value , (tb,'#', tb, simple_value)* 
48  >simple_value<       := string / number / name 
49  name                 := []-[a-z_A-Z!$&+./:;<>?^`|] , []-[a-z_A-Z0-9!$&+./:;<>?^`|]* 
50  number               :=  [0-9]+ 
51  string               := ('{' , braces_string?, '}') / ('"' , quotes_string?, '"') 
52  <braces_string>      := (-[{}"]+ / nested_string)+ 
53  <quotes_string>      := (-["{}]+ / nested_string)+ 
54  <nested_string>      := ('{' , braces_string, '}') / ('"' , quotes_string, '"') 
55  <junk>               := -[ \t\r\n]+ 
56  <tb>                 := (comment / ws)* 
57  <ws>                 := [ \t\n\r] 
58  <comment>            := '%' , -[\n]*, '\n' 
59  """
