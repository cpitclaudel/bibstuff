29  """bibgrammar module. 
30   
31     This module provides an EBNF description of the bibtex bibliography 
32     format. It draws largely from the grammar description in Nelson 
33     Beebe's Lex/Yacc parser (http://www.math.utah.edu/~beebe/) and also 
34     from Greg Ward's btOOL (http://starship.python.net/~gward/btOOL/) 
35     documentation.""" 
36   
37  __version__ = "$Revision: 1.5 $" 
38  __author__  =    '''Dylan W. Schwilk''' 
39   
40  from simpleparse.parser import Parser 
41  from simpleparse.common import numbers, strings, chartypes 
42  #import string 
43   
44   
45  # EBNF description of a bibtex file 
46  dec = r""" 
47  bibfile              := entry_or_junk* 
48  >entry_or_junk<      := (tb, object) / (tb, junk) 
49  >object<               := entry / macro / preamble / comment_entry 
50  entry                := '@', type , tb,  ( '{' , tb, contents, tb, '}' ) / ( '(' , tb, contents, tb, ')' ) 
51  macro                := '@', type, tb,  ( '{' , tb, macro_contents, tb, '}' ) / ( '(' , tb, macro_contents, tb, ')' ) 
52  preamble             := '@', type, tb,  ( '{' , tb, preamble_contents, tb, '}' ) / ( '(' , tb, preamble_contents, tb, ')' ) 
53  comment_entry        := '@', type, tb, string 
54  >contents<           := key , ',' , fields 
55  >macro_contents<     := fields 
56  >preamble_contents<  := value 
57  type                 := name 
58  key                  := number / name 
59  fields               := tb, field_comma+ , field? 
60  >field_comma<        := field , tb, ',' 
61  field                := tb, name, tb, '=' , tb, value 
62  value                := simple_value , (tb,'#', tb, simple_value)* 
63  >simple_value<       := string / number / name 
64  name                 := []-[a-z_A-Z!$&+./:;<>?^`|] , []-[a-z_A-Z0-9!$&+./:;<>?^`|]* 
65  number               :=  [0-9]+ 
66  string               := ('{' , braces_string?, '}') / ('"' , quotes_string?, '"') 
67  <braces_string>      := (-[{}"]+ / nested_string)+ 
68  <quotes_string>      := (-["{}]+ / nested_string)+ 
69  <nested_string>      := ('{' , braces_string, '}') / ('"' , quotes_string, '"') 
70  <junk>               := -[ \t\r\n]+ 
71  <tb>                 := (comment / ws)* 
72  <ws>                 := [ \t\n\r] 
73  <comment>            := '%' , -[\n]*, '\n' 
74  """ 
75   
76  ## The parsers 
77  parser = Parser(dec,'bibfile') 
78  entry_parser = Parser(dec,'entry')
