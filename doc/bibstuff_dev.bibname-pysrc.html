 37  name            := ( (tb, _and_) / (tb, part) / (tb , string) / (tb, comma) )+

 40  comma                := ','

 43  <lowercase>          := [a-z~'-]+  
 44  <two_part>           := [a-zA-Z~'-], [A-Z~'-]+ , [a-zA-Z~'-]+ 
 45  <braces_string>      := (-[{}]+ / nested_string)+ 
 46  <nested_string>      := ('{' , braces_string, '}')
